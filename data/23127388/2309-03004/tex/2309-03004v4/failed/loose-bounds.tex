\begin{lemma}[Bounds of on \sti{} transform]\label{lemma:bound_of_sti_old}
    Suppose $S \in \complexes$ satisfies 
    \begin{align}
        \frac{S}{1 + S} - z S = c + t,
    \end{align}
    where $z = 0 + v i \in \positivecomplex (v \in \reals^+), c \in \reals^+, t \in \complexes$, then
    \begin{align}
        \abs{\ipart{S}} \le \abs{S} \le&  \frac{2}{v} \left(c + \abs{t} + v + 1\right)^2 + 4 \left(c + \abs{t}\right).
    \end{align}
\end{lemma}
\begin{proof}
    $S$ is either
    \begin{align}
        S_1 = -\frac{-\sqrt{(-c-t-z+1)^2+4 z (-c-t)}+c+t+z-1}{2 z}
    \end{align}
    or
    \begin{align}
        S_2 = -\frac{\sqrt{(-c-t-z+1)^2+4 z (-c-t)}+c+t+z-1}{2 z},
    \end{align}
    both of which, since $z = v i, v \in \reals^+$, satisfy
    \begin{align}
        \abs{S} 
        \le&
            \frac{1}{v} \left(
                1 + v + \abs{t} + c + \abs{(-c-t-z+1)^2+4 z (-c-t)}
            \right)\\
        \le&
            \frac{1}{v} \left(
                1 + v + \abs{t} + c + \abs{-c-t-z+1}^2 + 4 v \abs{-c-t}
            \right)\\
        \le&
            \frac{1}{v} \left(
                1 + v + \abs{t} + c + \left(c + \abs{t} + v + 1\right)^2 + 4 v \left(c + \abs{t}\right)
            \right)\\
        \le&  \frac{2}{v} \left(c + \abs{t} + v + 1\right)^2 + 4 \left(c + \abs{t}\right).
    \end{align}
\end{proof}
